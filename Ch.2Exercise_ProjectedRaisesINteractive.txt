using System;

class ProjectedRaisesInteractive
{
    static void Main()
    {
        const double RAISE_RATE = 0.04;
        double salary1, salary2, salary3;

        Console.Write("Enter current salary for Employee 1: ");
        salary1 = Convert.ToDouble(Console.ReadLine());

        Console.Write("Enter current salary for Employee 2: ");
        salary2 = Convert.ToDouble(Console.ReadLine());

        Console.Write("Enter current salary for Employee 3: ");
        salary3 = Convert.ToDouble(Console.ReadLine());

        double newSalary1 = salary1 + (salary1 * RAISE_RATE);
        double newSalary2 = salary2 + (salary2 * RAISE_RATE);
        double newSalary3 = salary3 + (salary3 * RAISE_RATE);

        Console.WriteLine("\nProjected salaries for next year:");
        Console.WriteLine($"Employee 1: ${newSalary1: }");
        Console.WriteLine($"Employee 2: ${newSalary2: }");
        Console.WriteLine($"Employee 3: ${newSalary3: }");
    }
}
