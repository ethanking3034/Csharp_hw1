using System;

class MakeChange
{
    static void Main()
    {
        Console.Write("Enter an amount in dollars: ");
        bool isValid = int.TryParse(Console.ReadLine(), out int amount);

            int twenties = amount / 20;
            amount %= 20;

            int tens = amount / 10;
            amount %= 10;

            int fives = amount / 5;
            amount %= 5;

            int ones = amount;

            Console.WriteLine("Breakdown:");
            Console.WriteLine($"Twenties: {twenties}");
            Console.WriteLine($"Tens: {tens}");
            Console.WriteLine($"Fives: {fives}");
            Console.WriteLine($"Ones: {ones}");
     }
}
